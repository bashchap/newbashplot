#!/bin/bash
#
# Remember getopts?
#
#
source ~/dev/newbashplot/bin/setup.sh

typeset -i Opt1=${1:-81}
typeset -i Opt2=${2:-16}

typeset -i caaX=0 caaY=0 paaX=0 paaY=0 vbiX=0 vbiY=0 vbi=0 codePoint=0

# Pixel Addressable Area coordinates
paaX=Opt1
paaY=Opt2

# Calculate Cursor Addressable Area coordinates
caaX=paaX/CBC
caaY=paaY/CBR

# Calculate Virtual Bit Index coordinates (which pixel in a character block)
vbiX=paaX%CBC
vbiY=paaY%CBR

# Calculate Virtual Bitmap Index from vbiX/Y
vbi=$(((2**vbiY) << (vbiX*CBR)))

echo "paaX=$paaX"
echo "paaY=$paaY"
echo "caaX=$caaX"
echo "caaY=$caaY"
echo "vbiX=$vbiX"
echo "vbiY=$vbiY"
echo "vbi=$vbi"

codePoint=${VROBI[vbi]}
char=${uCodePoint[codePoint]}

echo "codePoint=$codepoint"
echo "char=$char"
